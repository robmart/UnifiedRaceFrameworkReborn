#############################
### URF Sapience Triggers ###
#############################
urf_is_low_sapient = {
    AND = {
        var:urf_sapience >= 50
        var:urf_sapience < 200
    }
}
urf_is_mitle_sapient = {
    AND = {
        var:urf_sapience >= 200
        var:urf_sapience < 750
    }
}
urf_is_ethereal_sapient = {
    AND = {
        var:urf_sapience >= 750
    }
}
